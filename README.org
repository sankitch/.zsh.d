My zsh configurations.

* Usage

#+BEGIN_EXAMPLE
cd; git clone git@github.com:sankitch/.zsh.d.git
cd .zsh.d
git submodule update --init
echo 'source $HOME/.zsh.d/zshenv' >~/.zshenv
echo 'source $HOME/.zsh.d/zshrc' >~/.zshrc
#+END_EXAMPLE

- canything
